First, I installed gfortran on Ubuntu 20.04 with
sudo apt-get install gfortran

which installs /usr/bin/gfortran

Next, I put the disort4 tar file in ~/PycharmProjects/planetary_disort and run
tar -xf disort4.0.98.tar.gz

and that makes the disort4.0.98 folder.

# Then to make the FORTRAN library in Python
disort_location = '/home/kyle/PycharmProjects/planetary_disort'
module_name = disort
python_path = '/home/kyle/PycharmProjects/planetary_disort/venv/bin/python3.8'

# These do the installation in Bash (though I made them Pythonic for future use)
cd '{}/disort4.0.98'.format(disort_location)
'{} -m numpy.f2py -c BDREF.f DISOBRDF.f DISORT.f ERRPACK.f LAPACK.f LINPAK.f RDI1MACH.f -m {}'.format(python_path, module_name)
cp '{}.cpython-38-x86_64-linux-gnu.so {}/site-packages'.format(module_name, python_path)
